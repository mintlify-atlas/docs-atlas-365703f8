---
title: "Consumer API"
description: "C API reference for consuming messages with librdkafka."
---

librdkafka provides two consumer interfaces: the high-level `KafkaConsumer` (recommended) that manages group membership and rebalancing, and the simple (legacy) consumer for manual partition management.

## High-level consumer

### Setup

```c consumer_setup.c
rd_kafka_conf_t *conf = rd_kafka_conf_new();
rd_kafka_conf_set(conf, "bootstrap.servers", "localhost:9092", errstr, sizeof(errstr));
rd_kafka_conf_set(conf, "group.id", "my-group", errstr, sizeof(errstr));
rd_kafka_conf_set(conf, "auto.offset.reset", "earliest", errstr, sizeof(errstr));

rd_kafka_t *rk = rd_kafka_new(RD_KAFKA_CONSUMER, conf, errstr, sizeof(errstr));
if (!rk) {
    fprintf(stderr, "Failed to create consumer: %s\n", errstr);
    return 1;
}

/* Redirect main queue to consumer queue */
rd_kafka_poll_set_consumer(rk);
```

<Note>
  `rd_kafka_poll_set_consumer()` must be called before subscribing. It redirects the main event queue so that `rd_kafka_consumer_poll()` serves all events including rebalance callbacks.
</Note>

### Subscribe and assign

```c
/* Subscribe to one or more topics (group-managed rebalancing) */
rd_kafka_resp_err_t rd_kafka_subscribe(rd_kafka_t *rk,
                                        const rd_kafka_topic_partition_list_t *topics);

/* Manually assign specific partitions (no group rebalancing) */
rd_kafka_resp_err_t rd_kafka_assign(rd_kafka_t *rk,
                                     const rd_kafka_topic_partition_list_t *partitions);

/* Remove subscription */
rd_kafka_resp_err_t rd_kafka_unsubscribe(rd_kafka_t *rk);

/* Get current subscription */
rd_kafka_resp_err_t rd_kafka_subscription(rd_kafka_t *rk,
                                           rd_kafka_topic_partition_list_t **topics);

/* Get current partition assignment */
rd_kafka_resp_err_t rd_kafka_assignment(rd_kafka_t *rk,
                                         rd_kafka_topic_partition_list_t **partitions);
```

Example subscription:

```c subscribe_example.c
rd_kafka_topic_partition_list_t *topics = rd_kafka_topic_partition_list_new(1);
rd_kafka_topic_partition_list_add(topics, "my-topic", RD_KAFKA_PARTITION_UA);

rd_kafka_resp_err_t err = rd_kafka_subscribe(rk, topics);
if (err) {
    fprintf(stderr, "Subscribe failed: %s\n", rd_kafka_err2str(err));
}
rd_kafka_topic_partition_list_destroy(topics);
```

### Poll loop

```c
rd_kafka_message_t *rd_kafka_consumer_poll(rd_kafka_t *rk, int timeout_ms);
```

`rd_kafka_consumer_poll()` returns a message or event within `timeout_ms` milliseconds. Returns `NULL` on timeout.

```c poll_loop.c
while (run) {
    rd_kafka_message_t *rkm = rd_kafka_consumer_poll(rk, 1000 /* 1 second */);
    if (!rkm)
        continue; /* Timeout, no message */

    if (rkm->err) {
        if (rkm->err == RD_KAFKA_RESP_ERR__PARTITION_EOF)
            fprintf(stdout, "Reached end of partition\n");
        else
            fprintf(stderr, "Consumer error: %s\n",
                    rd_kafka_message_errstr(rkm));
    } else {
        fprintf(stdout, "Message on %s [%"PRId32"] at offset %"PRId64": %.*s\n",
                rd_kafka_topic_name(rkm->rkt),
                rkm->partition, rkm->offset,
                (int)rkm->len, (char *)rkm->payload);
    }

    rd_kafka_message_destroy(rkm);
}
```

<Warning>
  Always call `rd_kafka_message_destroy()` when done with a message to free internal resources.
</Warning>

### Offset management

```c
/* Commit offsets for messages in partitions (async=1 for async, 0 for sync) */
rd_kafka_resp_err_t rd_kafka_commit(rd_kafka_t *rk,
                                     const rd_kafka_topic_partition_list_t *offsets,
                                     int async);

/* Commit the offset of a specific message (+1) */
rd_kafka_resp_err_t rd_kafka_commit_message(rd_kafka_t *rk,
                                              const rd_kafka_message_t *rkmessage,
                                              int async);

/* Retrieve committed offsets from broker */
rd_kafka_resp_err_t rd_kafka_committed(rd_kafka_t *rk,
                                        rd_kafka_topic_partition_list_t *partitions,
                                        int timeout_ms);

/* Get current fetch positions */
rd_kafka_resp_err_t rd_kafka_position(rd_kafka_t *rk,
                                       rd_kafka_topic_partition_list_t *partitions);
```

For at-least-once processing, disable auto-commit and commit after processing:

```c manual_commit.c
rd_kafka_conf_set(conf, "enable.auto.commit", "false", errstr, sizeof(errstr));

/* ... in your message loop, after processing: */
rd_kafka_commit_message(rk, rkm, 0 /* synchronous */);
```

### Seeking

```c
/* Seek all given partitions to specified offsets */
rd_kafka_error_t *rd_kafka_seek_partitions(rd_kafka_t *rk,
                                            rd_kafka_topic_partition_list_t *partitions,

```

### Incremental assign (COOPERATIVE rebalancing)

For COOPERATIVE assignors (incremental rebalancing), use these instead of `rd_kafka_assign()`:

```c
rd_kafka_error_t *rd_kafka_incremental_assign(rd_kafka_t *rk,
                                               const rd_kafka_topic_partition_list_t *partitions);

rd_kafka_error_t *rd_kafka_incremental_unassign(rd_kafka_t *rk,
                                                 const rd_kafka_topic_partition_list_t *partitions);
```

### Shutdown

```c shutdown.c
/* 1. Stop consuming and commit final offsets, leave the group */
rd_kafka_consumer_close(rk);

/* 2. Destroy the handle */
rd_kafka_destroy(rk);
```

<Note>
  `rd_kafka_consumer_close()` blocks for up to `session.timeout.ms`. All `rd_kafka_message_t` objects must be destroyed before calling it.
</Note>

## Simple (legacy) consumer

The simple consumer gives direct control over partition consumption without group management. It is considered legacy — prefer the high-level consumer for new applications.

```c simple_consumer.c
/* Start consuming from partition 0 at the beginning */
rd_kafka_topic_t *rkt = rd_kafka_topic_new(rk, "my-topic", NULL);
rd_kafka_consume_start(rkt, 0 /* partition */, RD_KAFKA_OFFSET_BEGINNING);

/* Poll for a message */
rd_kafka_message_t *rkm = rd_kafka_consume(rkt, 0, 1000 /* timeout_ms */);
if (rkm) {
    /* process message */
    rd_kafka_message_destroy(rkm);
}

/* Stop consuming */
rd_kafka_consume_stop(rkt, 0);
rd_kafka_topic_destroy(rkt);
```
